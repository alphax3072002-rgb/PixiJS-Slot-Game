# PROJECT SETUP

npm create vite@latest slot-game -- --template vanilla-ts   # create
cd slot-game                                                # enter
npm install                                                 # install

# DEPENDENCIES

npm install pixi.js@7                                       # pixi

# DEVELOPMENT

npm run dev                                                 # run

# VERSION CHECKS

# CLEAN & REINSTALL (if issues)

# BUILD (production)

npm run build                                               # build

# PREVIEW BUILD

npm run preview                                             # preview

# stop tracking node_modules

git rm -r --cached node_modules
